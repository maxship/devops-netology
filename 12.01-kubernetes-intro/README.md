    cores = 2
    memory = 4
  }

      image_id = "fd8mfc6omiki5govl68h" # Ubuntu-20.04

    }
  }

  }

    ssh-keys = "ubuntu:${file("~/.ssh/id_ed25519.pub")}" # SSH key
  }
